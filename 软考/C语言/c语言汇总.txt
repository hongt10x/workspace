#include <stdio.h>


void bubble_sort(int arr[], int len){

	int temp,i,j;
	for (i=0;i<len-1;i++){
	
		for(j=0;j<len-1-i;j++){
		
			if (arr[j]>arr[j+1]){
			
				temp = arr[j];
				arr[j] = arr[j+1];
				arr[j+1] = temp;
			}
		}
	}
	return;
}
	
int main () {
	int arr[] = {2,3,45,67,42,1,454,9,88,32};
	int i;
	bubble_sort(arr,8);
	for (i=0;i<10;i++){
	
		printf("%d\n",arr[i]);
	}
	return 0;
}

void swap1(int *p1,int *p2){

	int tmp = *p1;
	*p1 = *p2;
	*p2 = tmp;
	
}


void swap(int *p1,int *p2){

	int *tmp = p1;
	p1 = p2;
	p2 = tmp;
	cout <<"swap a地址"<<p1<<"  swap b地址"<<p2<<endl;
}


int main() {
	
	int a =10;
	int b=20;
	cout <<"a地址"<<&a<<"  b地址"<<&b<<endl;
	swap(&a,&b);
	cout <<"a="<<a<<"  b="<<b<<endl;


	return 0;
}
